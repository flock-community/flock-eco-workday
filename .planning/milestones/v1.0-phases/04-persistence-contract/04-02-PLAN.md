**BudgetAllocationPersistenceAdapter.kt** - Polymorphic adapter (reads + delete):
```kotlin
@Component
class BudgetAllocationPersistenceAdapter(
    private val repository: BudgetAllocationRepository,
) : BudgetAllocationPersistencePort {
    override fun findAllByPersonUuid(personUuid: UUID, year: Int) =
        repository.findAllByPersonUuidAndYear(personUuid, year)
            .map { it.toBudgetAllocationDomain() }

    override fun findAllByEventCode(eventCode: String) =
        repository.findAllByEventCode(eventCode)
            .map { it.toBudgetAllocationDomain() }

    override fun findById(id: Long) =
        repository.findByIdOrNull(id)?.toBudgetAllocationDomain()

    @Transactional
    override fun delete(id: Long) =
        repository.findByIdOrNull(id)?.run {
            val domain = toBudgetAllocationDomain()
            repository.delete(this)
            domain
        }
}
```
- Use `toBudgetAllocationDomain()` from mapper (polymorphic when/is dispatch)
- `@Transactional` on delete (find + delete in same transaction)
- Import `findByIdOrNull` from Spring Data

**Type-specific adapters** (HackTime, StudyTime, StudyMoney) - following CostExpensePersistenceAdapter pattern exactly:
```kotlin
@Component
class HackTimeBudgetAllocationPersistenceAdapter(
    private val repository: HackTimeBudgetAllocationRepository,
    private val entityManager: EntityManager,
) : HackTimeBudgetAllocationPersistencePort {
    @Transactional
    override fun create(allocation: HackTimeBudgetAllocation): HackTimeBudgetAllocation {
        val personReference = entityManager.getReference(Person::class.java, allocation.person.internalId)
        return repository.save(allocation.toEntity(personReference)).toDomain()
    }

    override fun findById(id: Long) = repository.findByIdOrNull(id)?.toDomain()

    @Transactional
    override fun updateIfExists(id: Long, allocation: HackTimeBudgetAllocation): HackTimeBudgetAllocation? {
        require(allocation.id == id) { "Cannot update allocation with different id" }
        return repository.existsById(id)
            .takeIf { it }
            ?.let {
                val personReference = entityManager.getReference(Person::class.java, allocation.person.internalId)
                repository.save(allocation.toEntity(personReference))
            }?.toDomain()
    }
}
```
- `EntityManager.getReference()` for Person (no extra SELECT)
- `@Transactional` on create and updateIfExists
- `require` precondition on update matching CostExpensePersistenceAdapter
- Same pattern for StudyTime and StudyMoney adapters
